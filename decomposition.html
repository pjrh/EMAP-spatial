decomp <- gva_prep %>%
  full_join(hours_prep, by = join_by(ITL_code, year)) %>%
  full_join(jobs_prep, by = join_by(ITL_code, year)) %>%
  full_join(pop_prep, by = join_by(ITL_code, year)) %>%
  full_join(whf_prep, by = join_by(ITL_code, year)) %>%
